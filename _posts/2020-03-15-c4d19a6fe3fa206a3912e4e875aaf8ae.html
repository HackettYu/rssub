1380. Lucky Numbers in a Matrix #Easy

题目解析

暴力搜索。先找每行最小的数字，再判断该数字是否为所在行最大的数字，满足条件则将其添加至 res 即可。

C++ 实现

class Solution {
public:
    vector<int> luckyNumbers (vector<vector<int>>& matrix) {
        vector<int> res;
        int m = matrix.size(), n = matrix[0].size();
        int tm = INT_MAX, tc = 0;
        bool flag = true;
        for (int i = 0; i < m; ++i) {
            tm = INT_MAX, tc = 0, flag = true;
            for (int j = 0; j < n; ++j) {
                if (matrix[i][j] < tm) {
                    tm = matrix[i][j];
                    tc = j;
                }
            }
            for (int k = 0; k < m; ++k) {
                if (matrix[k][tc] > tm) {
                    flag = false;
                    break;
                }
            }
            if (flag) res.push_back(tm);
        }
        return res;
    }
};

1381. Design a Stack With Increment Operation #Medium

题目解析

使用 vector 来模拟栈的操作即可。

C++ 实现

class CustomStack {
public:
    CustomStack(int maxSize) : ms(maxSize) { }
    
    void push(int x) {
        if (!full()) vec.push_back(x);
    }
    
    int pop() {
        int res = -1;
        if (!vec.empty()) {
            res = vec.back();
            vec.pop_back();
        }
        return res;
    }
    
    void increment(int k, int val) {
        for (int i = 0; i < k && i < vec.size(); ++i)
            vec[i] += val;
    }
    
    bool full() {
        return vec.size() == ms;
    }
private:
    int ms;
    vector<int> vec;
};

1382. Balance a Binary Search Tree #Medium

题目解析

首先，使用 helper1 通过中序遍历将二叉搜索树 root 转化为升序序列 v。

然后，使用 helper2 将升序序列 v 重建为平衡的二叉搜索树，返回根节点即可。

C++ 实现

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* balanceBST(TreeNode* root) {
        helper1(root);
        return helper2(0, v.size()-1);
    }
    void helper1(TreeNode *node) {
        if (!node) return;
        helper1(node->left);
        v.push_back(node->val);
        helper1(node->right);
    }
    TreeNode* helper2(int beg, int end) {
        if (beg > end) return nullptr;
        int mid = (beg + end) / 2;
        TreeNode *node = new TreeNode(v[mid]);
        node->left = helper2(beg, mid-1);
        node->right = helper2(mid+1, end);
        return node;
    }
private:
    vector<int> v;
};

1383. Maximum Performance of a Team #Hard

题目解析

首先，将工程师的效率和速度存入 vp，同时进行降序排序。

然后，遍历 vp，设第 i 个工程师的速度为 e且效率为 s，我们只从前 i 个工程师中挑人，那么第 i 个工程师的效率必然最小（因为 vp 降序），所以我们只要找出前 i 个工程师中速度最大的 k 个进行求和即可（也就是 Top K 问题，可以使用 priority_queue 实现最大堆）。

C++ 实现

class Solution {
public:
    int maxPerformance(int n, vector<int>& speed, vector<int>& efficiency, int k) {
        long long t = 1e9+7;
        vector<pair<int, int>> vp;
        for (int i = 0; i < n; ++i)
            vp.push_back({efficiency[i], speed[i]});
        sort(vp.rbegin(), vp.rend());
        priority_queue<int, vector<int>, greater<int>> pq;
        long long sum = 0, res = 0;
        for (int i = 0; i < n; ++i) {
            int e = vp[i].first, s = vp[i].second;
            pq.push(s);
            sum += s;
            if (pq.size() > k) {
                sum -= pq.top(); pq.pop();
            }
            res = max(res, sum*e);
        }
        return res % t;
    }
};
